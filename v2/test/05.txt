main()
eternal_sleep()

fn eternal_sleep() (
    loop()
)
fn allocate(dst, size) (
    set(addr(ret_sp), sub(get(3), 2))
    set(ret_sp, add(get(ret_sp), size))
    set(dst, ret_sp)
)
fn putc(ch) (
    write(1, addr(ch), 1)
)
fn print_endl() (
    putc(10)
)
fn print_int(x) (
    set(addr(m), 1000000000)
    if(x.lt(0)) (
        putc(45)
        set(addr(x), x.mul(-1))
    )
    if(x.eq(0)) (
        putc(48)
        return(0)
    )
    loop(
        if(x.div(m).eq(0).not()) (
            break
        )
        set(addr(m), m.div(10))
    )
    loop(
        if(m.eq(0)) (
            return(0)
        )
        putc(x.div(m).mod(10).add(48))
        set(addr(m), m.div(10))
    )
)
fn vec_init(v) (
    set(v, 0)
)
fn vec_push(v, x) (
    set(v, add(get(v), 1))
    set(add(v, get(v)), x)
)
fn vec_get(v, i) (
    return(get(add(add(v, 1), i)))
)

fn main() (
    set(addr(a_capacity), 10)

    allocate(addr(a), a_capacity)
    set(addr(status), vec_init(a))

    set(addr(i), 0)
    loop(
        if(i.eq(a_capacity)) (
            break
        )
        set(addr(status), a.vec_push(mul(i, 3)))
        set(addr(i), add(i, 1))
    )

    set(addr(i), 0)
    loop(
        if(i.eq(a_capacity)) (
            break
        )
        set(addr(status), print_int(a.vec_get(i)))
        set(addr(status), print_endl())
        set(addr(i), add(i, 1))
    )
)
