fn putc(x) (
    write(0, x.add(global_get(2)), 1)
)
fn endl() (

)
fn printi(x) (
    a.local_set(1000000)
    loop(
        tmp.local_set(putc(local_get(x).div(local_get(a)).mod(10).add(48)))
        a.local_set(local_get(a).div(10))
        if(local_get(a).eq(0)) break
    )
)
